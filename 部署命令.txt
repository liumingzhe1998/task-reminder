# 部署到 Render 的完整命令

# 第一步：打开 PowerShell 或命令提示符
# 按 Win+R，输入 powershell，回车

# 第二步：进入项目目录
cd C:\Users\12533\Desktop\实验\task-reminder

# 第三步：初始化 Git 仓库
git init

# 第四步：添加所有文件
git add .

# 第五步：提交代码
git commit -m "Initial commit: Task Reminder System"

# 第六步：在 GitHub 创建仓库
# 1. 访问 https://github.com/new
# 2. 仓库名称：task-reminder
# 3. 选择 Public
# 4. 点击 Create repository

# 第七步：添加远程仓库（替换 YOUR_USERNAME 为你的 GitHub 用户名）
git remote add origin https://github.com/YOUR_USERNAME/task-reminder.git

# 第八步：推送到 GitHub
git branch -M main
git push -u origin main

# 如果需要密码，使用 GitHub Personal Access Token，不是登录密码
